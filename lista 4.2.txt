SELECT
imie,
Count(*) as [ilosc]
FROM
gracze
GROUP BY
imie
HAVING
Count(*) > 10;

stan,

FROM
oceny
GROUP BY
stan
HAVING
Count(*) > 1100;







----Zadanie 3
SELECT
id_gry,
MIN(ocena) as [najnizsze]
FROM
oceny
GROUP BY
id_gry
HAVING
MIN(ocena)> 1;








----Zadanie 4
SELECT
nazwisko,
Count(*) as [ilosc (parzyste)]
FROM
gracze
GROUP BY
nazwisko
HAVING
Count(*) MOD 2 = 0;






----Zadanie 5
SELECT
id_gracza,
SUM(ocena) as [suma_ocen]
FROM
oceny
GROUP BY
id_gracza
HAVING
SUM(ocena) LIKE "##"
ORDER BY SUM(ocena) ASC;
